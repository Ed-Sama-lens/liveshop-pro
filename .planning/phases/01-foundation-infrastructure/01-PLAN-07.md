    expect(result.meta).toBeNull();

  it('includes meta when provided', () => {
    const meta = { page: 1, pageSize: 10, total: 100, totalPages: 10 };
    const result = ok([1, 2, 3], meta);
    expect(result.meta).toEqual(meta);
  });

    const input = { id: 1 };
    const result = ok(input);
    expect(result).not.toBe(input);
    expect(result.data).not.toBe(result);
  });
});

  it('returns success: false with error message', () => {

import {
  AppError, ValidationError, AuthError, ForbiddenError,
  NotFoundError, ConflictError, RateLimitError, toAppError,
} from '@/lib/errors/index';

describe('AppError hierarchy', () => {
  it('ValidationError has status 400 and correct code', () => {
    const err = new ValidationError('bad input');
    expect(err.status).toBe(400);
    expect(err.code).toBe('VALIDATION_ERROR');
    expect(err instanceof AppError).toBe(true);
  });

  it('AuthError has status 401', () => {
    expect(new AuthError().status).toBe(401);
  });

  it('ForbiddenError has status 403', () => {
    expect(new ForbiddenError().status).toBe(403);
  });

  it('NotFoundError includes resource name in message', () => {
    const err = new NotFoundError('User');
    expect(err.message).toBe('User not found');
    expect(err.status).toBe(404);
  });

  it('ConflictError has status 409', () => {
    expect(new ConflictError('duplicate').status).toBe(409);
  });

  it('RateLimitError has status 429', () => {
    expect(new RateLimitError().status).toBe(429);
  });

  it('ValidationError fields is a copy — mutating it does not affect source', () => {
    const fields = { email: ['required'] };
    const err = new ValidationError('bad', fields);
    err.fields['email']?.push('mutated');
    expect(fields['email']).toEqual(['required']);
  });

  it('all subclasses pass instanceof AppError', () => {
    const errors = [
      new ValidationError('v'), new AuthError(), new ForbiddenError(),
      new NotFoundError('X'), new ConflictError('c'), new RateLimitError(),
    ];
    for (const err of errors) {
      expect(err instanceof AppError).toBe(true);
    }
  });
});

describe('toAppError()', () => {
  it('returns AppError unchanged', () => {
    const original = new ForbiddenError('denied');
    expect(toAppError(original)).toBe(original);
  });

  it('converts plain Error to generic AppError — does not leak internal message', () => {
    const err = toAppError(new Error('internal stack trace with db details'));
    expect(err instanceof AppError).toBe(true);
    expect(err.message).not.toContain('internal stack trace');
    expect(err.message).toBe('An unexpected error occurred');
    expect(err.status).toBe(500);
  });

  it('converts unknown value to generic AppError', () => {
    const err = toAppError('some string error');
    expect(err instanceof AppError).toBe(true);
    expect(err.status).toBe(500);
  });
});
