```typescript
/**
 * Room naming conventions for Socket.IO multi-tenancy.
 * All shop events must use shop:{shopId} rooms — never global rooms.
 *
 * Conventions enforced:
 * - shopRoom(shopId) generates the canonical room name
 * - isValidRoom validates that a room name follows the shop:{id} pattern
 * - getRoomName extracts the shopId from a room name
 */

const SHOP_ROOM_PREFIX = 'shop:' as const;

/**
 * Generate the canonical room name for a shop.
 * All emitToRoom calls and socket joins must use this.
 */
export function shopRoom(shopId: string): string {
  return `${SHOP_ROOM_PREFIX}${shopId}`;
}

/**
 * Check if a room name is a valid shop room (not a global broadcast room).
 * Used to validate that no code emits to global rooms by accident.
 */
export function isValidRoom(roomName: string): boolean {
  return roomName.startsWith(SHOP_ROOM_PREFIX) && roomName.length > SHOP_ROOM_PREFIX.length;
}

/**
 * Extract the shopId from a room name.
 * Returns null if the room name is not a valid shop room.
 */
export function getRoomName(roomName: string): string | null {
  if (!isValidRoom(roomName)) return null;
  return roomName.slice(SHOP_ROOM_PREFIX.length);
}

export { SHOP_ROOM_PREFIX };
```

```typescript
/**
 * Type-safe job data contracts for all Bull queues.
 * Job producers (API routes, webhooks) must conform to these types.
 * Job processors (in src/server/queues/*.processor.ts) receive these types.
 *
 * Import from this file, not from individual queue files.
 */

// ─── Order Queue ─────────────────────────────────────────────────────────────

export type OrderJobAction =
  | 'order:created'
  | 'order:confirm'
  | 'order:cancel'
  | 'stock:reserve'
  | 'stock:release'
  | 'stock:reservation-expire';

export interface OrderJobData {
  readonly shopId: string;
  readonly orderId: string;
  readonly action: OrderJobAction;
  readonly metadata?: Record<string, unknown>;
}

// ─── Message Queue ────────────────────────────────────────────────────────────

export type MessageJobAction =
  | 'message:send'
  | 'message:receive'
  | 'auto-reply:check'
  | 'message:mark-read';

export type MessageDirection = 'INBOUND' | 'OUTBOUND';

export interface MessageJobData {
  readonly shopId: string;
  readonly chatId: string;
  readonly direction: MessageDirection;
  readonly action: MessageJobAction;
  readonly content?: string;
  readonly fbMessageId?: string;
}

// ─── Inventory Queue ──────────────────────────────────────────────────────────

export type InventoryJobAction =
  | 'inventory:update'
  | 'stock:alert'
  | 'reservation:expire'
  | 'sync:facebook-catalog';

export interface InventoryJobData {
  readonly shopId: string;
  readonly variantId: string;
  readonly quantityDelta: number;
  readonly action: InventoryJobAction;
  readonly reason?: string;
}

// ─── Analytics Queue ──────────────────────────────────────────────────────────

export type AnalyticsJobAction =
  | 'session:update-counters'
  | 'session:viewer-count'
  | 'report:generate'
  | 'metrics:aggregate';

export interface AnalyticsJobData {
  readonly shopId: string;
  readonly sessionId: string;
  readonly event: AnalyticsJobAction;
  readonly payload?: Record<string, unknown>;
}

// ─── Union type for type guards ────────────────────────────────────────────────

export type AnyJobData =
  | OrderJobData
  | MessageJobData
  | InventoryJobData
  | AnalyticsJobData;

/**
 * Type guard for OrderJobData.
 */
export function isOrderJob(data: AnyJobData): data is OrderJobData {
  return 'orderId' in data && 'action' in data;
}

/**
 * Type guard for MessageJobData.
 */
export function isMessageJob(data: AnyJobData): data is MessageJobData {
  return 'chatId' in data && 'direction' in data;
}

/**
 * Type guard for InventoryJobData.
 */
export function isInventoryJob(data: AnyJobData): data is InventoryJobData {
  return 'variantId' in data && 'quantityDelta' in data;
}

/**
 * Type guard for AnalyticsJobData.
 */
export function isAnalyticsJob(data: AnyJobData): data is AnalyticsJobData {
  return 'sessionId' in data && 'event' in data;
}
```
