Create `src/lib/db/queues.ts` — four Bull queues backed by Redis. Import in server-side code only, never in Client Components:

```typescript
import Bull from 'bull';
import { env } from '@/lib/env';

const redisConfig = { redis: env.REDIS_URL };

const defaultJobOptions: Bull.JobOptions = {
  attempts: 3,
  backoff: { type: 'exponential', delay: 1000 },
  removeOnComplete: 100,
  removeOnFail: 500,
};

/** Order processing: order:created, order:confirm, order:cancel, stock:reserve, stock:release */
export const orderQueue = new Bull('orders', { ...redisConfig, defaultJobOptions });

/** Messenger messages: message:send, message:receive, auto-reply:check */
export const messageQueue = new Bull('messages', { ...redisConfig, defaultJobOptions });

/** Inventory sync: inventory:update, stock:alert, reservation:expire */
export const inventoryQueue = new Bull('inventory', { ...redisConfig, defaultJobOptions });

/** Analytics aggregation: session:update-counters, report:generate */
export const analyticsQueue = new Bull('analytics', { ...redisConfig, defaultJobOptions });

export const allQueues = [orderQueue, messageQueue, inventoryQueue, analyticsQueue] as const;

export type QueueName = 'orders' | 'messages' | 'inventory' | 'analytics';
```

Create `src/lib/db/rls.ts` — PostgreSQL RLS setup SQL + Prisma shopId middleware. Defense in depth: middleware is app-level guard, RLS is database-level guard:

```typescript
import { prisma } from './prisma';
import { logger } from '@/lib/logging/logger';

// Models with a direct shopId column. ProductVariant inherits shop scope
// via Product relation — do NOT include it here (it has no shopId field).
const SHOP_SCOPED_MODELS = [
  'Product', 'ProductCategory',
  'Customer', 'Order', 'Chat', 'ChatMessage', 'LiveSession',
  'StorefrontProduct', 'Cart',
] as const;

type ShopScopedModel = typeof SHOP_SCOPED_MODELS[number];

export const RLS_SETUP_SQL = `
-- Tables with direct shopId column
ALTER TABLE "Product" ENABLE ROW LEVEL SECURITY;
ALTER TABLE "ProductCategory" ENABLE ROW LEVEL SECURITY;
ALTER TABLE "Customer" ENABLE ROW LEVEL SECURITY;
ALTER TABLE "Order" ENABLE ROW LEVEL SECURITY;
ALTER TABLE "Chat" ENABLE ROW LEVEL SECURITY;
ALTER TABLE "ChatMessage" ENABLE ROW LEVEL SECURITY;
ALTER TABLE "LiveSession" ENABLE ROW LEVEL SECURITY;
ALTER TABLE "StorefrontProduct" ENABLE ROW LEVEL SECURITY;
ALTER TABLE "Cart" ENABLE ROW LEVEL SECURITY;

-- Tables without shopId (inherit via parent relation) — RLS enabled but no direct policy

export function createShopMiddleware(): Parameters<typeof prisma.$use>[0] {
  return async (params, next) => {
    const writeOps = ['create', 'createMany', 'update', 'updateMany', 'upsert'];
    const isShopScoped = SHOP_SCOPED_MODELS.includes(params.model as ShopScopedModel);
    const isWrite = writeOps.includes(params.action);

    if (isShopScoped && isWrite) {
      const data = params.args?.data as Record<string, unknown> | undefined;
      if (data && !data['shopId']) {
        throw new Error(
          `[RLS] shopId is required for ${params.model}.${params.action}. ` +
          'All shop-scoped writes must include shopId.'
        );
      }
    }

    // Never mutate params — pass through as-is
    const result: unknown = await next(params);
    return result;
  };
}
```

```typescript
import { createCipheriv, createDecipheriv, randomBytes } from 'crypto';
import { env } from '@/lib/env';

const ALGORITHM = 'aes-256-gcm';
const IV_LENGTH = 16;
const TAG_LENGTH = 16;

/**
 * Encrypt a Facebook access token for storage.
 * Returns iv:authTag:ciphertext (all hex-encoded).
 * Each call produces different output due to random IV.
 */
export function encryptToken(plaintext: string): string {
  const key = getEncryptionKey();
  const iv = randomBytes(IV_LENGTH);
  const cipher = createCipheriv(ALGORITHM, key, iv);

  const encrypted = Buffer.concat([cipher.update(plaintext, 'utf8'), cipher.final()]);
  const tag = cipher.getAuthTag();

  return [iv.toString('hex'), tag.toString('hex'), encrypted.toString('hex')].join(':');
}

/**
 * Decrypt a stored Facebook token.
 * Throws if ciphertext is tampered (auth tag mismatch — GCM authentication).
 */
export function decryptToken(stored: string): string {
  const key = getEncryptionKey();
  const parts = stored.split(':');

  if (parts.length !== 3) {
    throw new Error('Invalid encrypted token format — expected iv:tag:ciphertext');
  }

  const [ivHex, tagHex, encryptedHex] = parts as [string, string, string];
  const iv = Buffer.from(ivHex, 'hex');
  const tag = Buffer.from(tagHex, 'hex');
  const encrypted = Buffer.from(encryptedHex, 'hex');

  const decipher = createDecipheriv(ALGORITHM, key, iv);
  decipher.setAuthTag(tag);

  const decrypted = Buffer.concat([decipher.update(encrypted), decipher.final()]);
  return decrypted.toString('utf8');
}

function getEncryptionKey(): Buffer {
  const keyHex = env.TOKEN_ENCRYPTION_KEY;
  if (!keyHex) {
    throw new Error(
      'TOKEN_ENCRYPTION_KEY is required for token encryption. ' +
      'Generate one with: openssl rand -hex 32'
    );
  }
  return Buffer.from(keyHex, 'hex');
}
```
