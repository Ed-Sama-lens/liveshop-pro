Create `src/lib/auth/auth.ts`. JWT maxAge is 24 hours (reduced from 30 days per security review — shorter window limits stolen token exposure). On successful sign-in, writes a LoginEvent audit record:

```typescript
import NextAuth from 'next-auth';
import FacebookProvider from 'next-auth/providers/facebook';
import { PrismaAdapter } from '@auth/prisma-adapter';
import { prisma } from '@/lib/db/prisma';
import { logger } from '@/lib/logging/logger';
import { env } from '@/lib/env';
import type { UserRole } from './types';

export const { handlers, auth, signIn, signOut } = NextAuth({
  adapter: PrismaAdapter(prisma),
  providers: [
    FacebookProvider({
      clientId: env.FACEBOOK_CLIENT_ID,
      clientSecret: env.FACEBOOK_CLIENT_SECRET,
      authorization: {
        params: { scope: 'email,public_profile,pages_show_list,pages_messaging' },
      },
    }),
  ],
  session: {
    strategy: 'jwt',
    maxAge: 24 * 60 * 60, // 24 hours — reduced from 30 days per security review [REVIEW-FIX]
  },
  callbacks: {
    async signIn({ user, account }) {
      // Write session audit log — never throw on failure, just log
      try {
        if (user.id && account?.provider === 'facebook') {
          await prisma.loginEvent.create({
            data: { userId: user.id, success: true },
          });
        }
      } catch (err) {
        logger.error({ err, userId: user.id }, '[Auth] Failed to write LoginEvent');
      }
      return true;
    },

    async jwt({ token, user }) {
      if (user) {
        const dbUser = await prisma.user.findUnique({
          where: { id: user.id },
          select: { role: true },
        });
        // Return new token object — never mutate
        return {
          ...token,
          id: user.id,
          role: dbUser?.role ?? ('CHAT_SUPPORT' as UserRole),
          shopId: null as string | null,
        };
      }
      return token;
    },

    async session({ session, token }) {
      // Return new session object — do not mutate
      return {
        ...session,
        user: {
          ...session.user,
          id: token.id as string,
          role: token.role as UserRole,
          shopId: (token.shopId as string | null) ?? null,
        },
      };
    },
  },
  pages: {
    signIn: '/auth/sign-in',
    error: '/auth/error',
  },
  logger: {
    error(code, metadata) { logger.error({ code, metadata }, '[NextAuth] Error'); },
    warn(code) { logger.warn({ code }, '[NextAuth] Warning'); },
    debug(code, metadata) { logger.debug({ code, metadata }, '[NextAuth] Debug'); },
  },
});
```

Create `src/lib/auth/session.ts` with server-side session helpers. requireShopMember validates active shop membership — prevents privilege escalation by verifying the user belongs to the requested shop:

```typescript
import { auth } from './auth';
import { redirect } from 'next/navigation';
import { prisma } from '@/lib/db/prisma';
import type { UserRole } from './types';

export async function getSession() {
  return auth();
}

/**
 * Get the current user or redirect to sign-in.
 * Verifies user is still active in the database (prevents access after deactivation).
 * [REVIEW-FIX: P0 #2]
 */
export async function requireAuth() {
  const session = await auth();

  if (!session?.user) {
    redirect('/auth/sign-in');
  }

  const dbUser = await prisma.user.findUnique({
    where: { id: session.user.id, isActive: true },
    select: { id: true, role: true, isActive: true },
  });

  if (!dbUser) {
    redirect('/auth/sign-in');
  }

  return session.user;
}

/**
 * Get the current user and verify active shop membership.
 * Ensures session user actually belongs to the requested shop.
 * Prevents horizontal privilege escalation across shops.
 * [REVIEW-FIX: P0 #2]
 */
export async function requireShopMember(shopId: string) {
  const user = await requireAuth();

  const membership = await prisma.shopMember.findUnique({
    where: { shopId_userId: { shopId, userId: user.id } },
    select: { role: true, shopId: true },
  });

  if (!membership) {
    redirect('/unauthorized');
  }

  return { ...user, shopId: membership.shopId, role: membership.role };
}

/**
 * Get the current user and verify role.
 * Redirects to /unauthorized if role not permitted.
 */
export async function requireRole(allowedRoles: UserRole[]) {
  const user = await requireAuth();
  if (!allowedRoles.includes(user.role)) redirect('/unauthorized');
  return user;
}

/**
 * Return true if the user has at least one of the given roles.
 * Pure function — does not redirect, no side effects.
 */
export function hasRole(userRole: UserRole, allowedRoles: UserRole[]): boolean {
  return allowedRoles.includes(userRole);
}
```

Create `src/lib/validation/middleware.ts` — three utilities: rate limiter for auth routes, CORS header helper, Zod body/query validation wrapper. All functions return new objects — never mutate the request:

```typescript
import { NextResponse } from 'next/server';
import type { NextRequest } from 'next/server';
import { RateLimiterMemory } from 'rate-limiter-flexible';
import { z, type ZodSchema } from 'zod';
import { error } from '@/lib/api/response';
import { RateLimitError, ValidationError } from '@/lib/errors/index';
import { env } from '@/lib/env';

// ─── Rate Limiting ─────────────────────────────────────────────────────────

/**
 * In-memory rate limiter for authentication endpoints.
 * Default: 20 requests per 15 minutes per IP.
 * For multi-instance deployments, swap to RateLimiterRedis.
 * [REVIEW-FIX: P1 #8]
 */
const authRateLimiter = new RateLimiterMemory({
  points: env.RATE_LIMIT_MAX,
  duration: env.RATE_LIMIT_WINDOW_MS / 1000,
});

/**
 * Apply rate limiting to an API route handler.
 * Returns null if under limit (proceed), or a 429 NextResponse if exceeded.
 *
 * Usage:
 *   const limited = await withRateLimit(request);
 *   if (limited) return limited;
 */
export async function withRateLimit(request: NextRequest): Promise<NextResponse | null> {
  const ip =
    request.headers.get('x-forwarded-for')?.split(',')[0]?.trim() ??
    request.headers.get('x-real-ip') ??
    '127.0.0.1';

  try {
    await authRateLimiter.consume(ip);
    return null;
  } catch {
    const rateLimitErr = new RateLimitError();
    return NextResponse.json(
      error(rateLimitErr.message, rateLimitErr.code),
      {
        status: 429,
        headers: { 'Retry-After': String(Math.ceil(env.RATE_LIMIT_WINDOW_MS / 1000)) },
      }
    );
  }
}

// ─── CORS ──────────────────────────────────────────────────────────────────

/**
 * Returns CORS headers allowing only the configured app origin.
 * NEVER returns wildcard (*) — specific origin only for security.
 * [REVIEW-FIX: P2 #12]
 */
export function getCorsHeaders(): HeadersInit {
  return {
    'Access-Control-Allow-Origin': env.NEXT_PUBLIC_APP_URL,
    'Access-Control-Allow-Methods': 'GET, POST, PUT, PATCH, DELETE, OPTIONS',
    'Access-Control-Allow-Headers': 'Content-Type, Authorization',
    'Access-Control-Allow-Credentials': 'true',
    'Access-Control-Max-Age': '86400',
  };
}

/**
 * Handle CORS preflight OPTIONS requests.
 */
export function handleCorsPreFlight(): NextResponse {
  return new NextResponse(null, { status: 204, headers: getCorsHeaders() });
}

// ─── Zod Validation ────────────────────────────────────────────────────────

export interface ValidationResult<T> {
  readonly success: true;
  readonly data: T;
}

export interface ValidationFailure {
  readonly success: false;
  readonly response: NextResponse;
}

/**
 * Validate the JSON body of a request against a Zod schema.
 * Returns validated data on success or a 400 NextResponse on failure.
 * Never mutates the request.
 *
 * Usage:
 *   const result = await validateBody(request, MySchema);
 *   if (!result.success) return result.response;
 *   const { name } = result.data;
 */
export async function validateBody<T>(
  request: NextRequest,
  schema: ZodSchema<T>
): Promise<ValidationResult<T> | ValidationFailure> {
  let body: unknown;

  try {
    body = await request.json();
  } catch {
    return {
      success: false,
      response: NextResponse.json(
        error('Request body must be valid JSON', 'INVALID_JSON'),
        { status: 400 }
      ),
    };
  }

  const result = schema.safeParse(body);

  if (!result.success) {
    const fields: Record<string, string[]> = {};
    for (const issue of result.error.errors) {
      const key = issue.path.join('.') || '_root';
      fields[key] = [...(fields[key] ?? []), issue.message];
    }
    const validationErr = new ValidationError('Validation failed', fields);
    return {
      success: false,
      response: NextResponse.json(
        { ...error(validationErr.message, validationErr.code), fields },
        { status: 400 }
      ),
    };
  }

  return { success: true, data: result.data };
}

/**
 * Validate URL search params against a Zod schema.
 * Returns validated data or a 400 response.
 */
export function validateQuery<T>(
  request: NextRequest,
  schema: ZodSchema<T>
): ValidationResult<T> | ValidationFailure {
  const params = Object.fromEntries(request.nextUrl.searchParams.entries());
  const result = schema.safeParse(params);

  if (!result.success) {
    const fields: Record<string, string[]> = {};
    for (const issue of result.error.errors) {
      const key = issue.path.join('.') || '_root';
      fields[key] = [...(fields[key] ?? []), issue.message];
    }
    const validationErr = new ValidationError('Invalid query parameters', fields);
    return {
      success: false,
      response: NextResponse.json(
        { ...error(validationErr.message, validationErr.code), fields },
        { status: 400 }
      ),
    };
  }

  return { success: true, data: result.data };
}
```
